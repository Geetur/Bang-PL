# solution to two sum

fn two_sum args

    target = args[0]
    input = args[1]

    for range1 [0, len{input} - 1]
        for range2 [range1 + 1, len{input}]
            if input[range1] + input[range2] == target
                print{input[range1], input[range2]}
            end
        end
    end
end
